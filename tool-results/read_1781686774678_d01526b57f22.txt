    75→// ─── M3U Parser ────────────────────────────────────────────────────────────────
    76→
    77→function parseM3U(text: string): Channel[] {
    78→  const lines = text.split('\n').map(l => l.trim()).filter(Boolean);
    79→  const parsed: Channel[] = [];
    80→  let current: Partial<Channel> | null = null;
    81→  let nextId = 1000;
    82→
    83→  for (const line of lines) {
    84→    if (line.startsWith('#EXTINF:')) {
    85→      current = {};
    86→      const logoMatch = line.match(/tvg-logo="([^"]*)"/);
    87→      const groupMatch = line.match(/group-title="([^"]*)"/);
    88→      const nameMatch = line.match(/,(.+)$/);
    89→      if (logoMatch) current.logo = logoMatch[1];
    90→      if (groupMatch) current.category = groupMatch[1];
    91→      if (nameMatch) current.name = nameMatch[1].trim();
    92→    } else if (line.startsWith('#')) {
    93→      continue;
    94→    } else if (current) {
    95→      current.url = line;
    96→      current.id = nextId++;
    97→      if (!current.category) current.category = 'Imported';
    98→      if (!current.name) current.name = `Channel ${current.id}`;
    99→      if (!current.logo) current.logo = '';
   100→      parsed.push(current as Channel);
   101→      current = null;
   102→    }
   103→  }
   104→  return parsed;
   105→}

   109→export default function Home() {
   110→  const [channels, setChannels] = useState<Channel[]>(defaultChannels);
   111→  const [filteredChannels, setFilteredChannels] = useState<Channel[]>(defaultChannels);
   112→  const [currentChannel, setCurrentChannel] = useState<Channel | null>(null);
   113→  const [sidebarOpen, setSidebarOpen] = useState(true);
   114→  const [searchQuery, setSearchQuery] = useState('');
   115→  const [activeCategory, setActiveCategory] = useState<Category>('All');
   116→  const [favorites, setFavorites] = useState<number[]>(() => {
   117→    if (typeof window === 'undefined') return [];
   118→    const saved = localStorage.getItem('iptv_favorites');
   119→    return saved ? JSON.parse(saved) : [];
   120→  });
   121→  const [volume, setVolume] = useState(() => {
   122→    if (typeof window === 'undefined') return 1;
   123→    const saved = localStorage.getItem('iptv_volume');
   124→    return saved ? parseFloat(saved) : 1;
   125→  });
   126→  const [muted, setMuted] = useState(false);
   127→  const [isPlaying, setIsPlaying] = useState(false);
   128→  const [isLoading, setIsLoading] = useState(false);
   129→  const [error, setError] = useState<string | null>(null);
   130→  const [showControls, setShowControls] = useState(true);
   131→  const [showOSD, setShowOSD] = useState(false);
   132→  const [isDark, setIsDark] = useState(true);
   133→  const [isOnline, setIsOnline] = useState(true);
   134→  const [isPiP, setIsPiP] = useState(false);
   135→  const [isFullscreen, setIsFullscreen] = useState(false);
   136→  const [clock, setClock] = useState('');
   137→  const [showImport, setShowImport] = useState(false);
   138→  const [showHelp, setShowHelp] = useState(false);
   139→
   140→  const videoRef = useRef<HTMLVideoElement>(null);
   141→  const containerRef = useRef<HTMLDivElement>(null);
   142→  const hlsRef = useRef<Hls | null>(null);
   143→  const controlsTimeoutRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   144→  const osdTimeoutRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   145→  const fileInputRef = useRef<HTMLInputElement>(null);
   146→  const channelListRef = useRef<HTMLDivElement>(null);
   147→
   148→  // ─── Dark mode ───────────────────────────────────────────────────────────────
   149→
   150→  useEffect(() => {
   151→    document.documentElement.classList.toggle('dark', isDark);
   152→  }, [isDark]);
   153→
   154→  // ─── Clock ───────────────────────────────────────────────────────────────────
   155→
   156→  useEffect(() => {
   157→    const update = () => {
   158→      const now = new Date();
   159→      setClock(now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }));
   160→    };
   161→    update();
   162→    const interval = setInterval(update, 1000);
   163→    return () => clearInterval(interval);
   164→  }, []);
   165→
   166→  // ─── Online/Offline ──────────────────────────────────────────────────────────
   167→
   168→  useEffect(() => {
   169→    const on = () => setIsOnline(true);
   170→    const off = () => setIsOnline(false);
   171→    window.addEventListener('online', on);
   172→    window.addEventListener('offline', off);
   173→    return () => { window.removeEventListener('online', on); window.removeEventListener('offline', off); };
   174→  }, []);
   175→
   176→  // ─── Filter channels ────────────────────────────────────────────────────────
   177→
   178→  useEffect(() => {
   179→    const filtered = channels.filter(channel => {
   180→      const matchesSearch = channel.name.toLowerCase().includes(searchQuery.toLowerCase());
   181→      let matchesCategory = activeCategory === 'All';
   182→      if (activeCategory === 'Favorites') {
   183→        matchesCategory = favorites.includes(channel.id);
   184→      } else {
   185→        matchesCategory = channel.category === activeCategory;
   186→      }
   187→      return matchesSearch && matchesCategory;
   188→    });
   189→    setFilteredChannels(filtered);
   190→  }, [channels, searchQuery, activeCategory, favorites]);
   191→
   192→  // ─── Play channel ────────────────────────────────────────────────────────────
   193→
   194→  const playChannel = useCallback((channel: Channel) => {
   195→    if (!videoRef.current || !channel.url) return;
   196→    setCurrentChannel(channel);
   197→    setIsLoading(true);
   198→    setError(null);
   199→    setShowOSD(true);
   200→    localStorage.setItem('iptv_last_channel', String(channel.id));
   201→
   202→    if (hlsRef.current) {
   203→      hlsRef.current.destroy();
   204→      hlsRef.current = null;
   205→    }
   206→
   207→    const video = videoRef.current;
   208→
   209→    if (channel.url.includes('.m3u8')) {
   210→      if (Hls.isSupported()) {
   211→        const hls = new Hls({
   212→          enableWorker: true,
   213→          lowLatencyMode: true,
   214→          maxBufferLength: 30,
   215→          maxMaxBufferLength: 60,
   216→        });
   217→        hlsRef.current = hls;
   218→        hls.loadSource(channel.url);
   219→        hls.attachMedia(video);
   220→        hls.on(Hls.Events.MANIFEST_PARSED, () => {
   221→          video.play().catch(() => {});
   222→        });
   223→        hls.on(Hls.Events.ERROR, (_event, data) => {
   224→          if (data.fatal) {
   225→            switch (data.type) {
   226→              case Hls.ErrorTypes.NETWORK_ERROR:
   227→                hls.startLoad();
   228→                break;
   229→              case Hls.ErrorTypes.MEDIA_ERROR:
   230→                hls.recoverMediaError();
   231→                break;
   232→              default:
   233→                setError(`Stream load failed: ${data.type}`);
   234→                setIsLoading(false);
   235→                break;
   236→            }
   237→          }
   238→        });
   239→      } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
   240→        video.src = channel.url;
   241→        video.play().catch(() => {});
   242→      }
   243→    } else {
   244→      video.src = channel.url;
   245→      video.play().catch(() => {});
   246→    }
   247→
   248→    if (osdTimeoutRef.current) clearTimeout(osdTimeoutRef.current);
   249→    osdTimeoutRef.current = setTimeout(() => setShowOSD(false), 3000);
   250→  }, []);
   251→
   252→  // ─── Video events ────────────────────────────────────────────────────────────
   253→
   254→  useEffect(() => {
   255→    const video = videoRef.current;
   256→    if (!video) return;
   257→    const handlers: Record<string, () => void> = {
   258→      play: () => setIsPlaying(true),
   259→      pause: () => setIsPlaying(false),
   260→      canplay: () => setIsLoading(false),
   261→      waiting: () => setIsLoading(true),
   262→      playing: () => setIsLoading(false),
   263→      error: () => { setError('Stream playback failed'); setIsLoading(false); },
   264→    };
   265→    Object.entries(handlers).forEach(([event, handler]) => video.addEventListener(event, handler));
   266→    return () => { Object.entries(handlers).forEach(([event, handler]) => video.removeEventListener(event, handler)); };
   267→  }, []);
   268→
   269→  // ─── Volume sync ────────────────────────────────────────────────────────────
   270→
   271→  useEffect(() => {
   272→    if (videoRef.current) {
   273→      videoRef.current.volume = volume;
   274→      videoRef.current.muted = muted;
   275→    }
   276→    localStorage.setItem('iptv_volume', String(volume));
   277→  }, [volume, muted]);
   278→
   279→  // ─── Auto-play last channel ─────────────────────────────────────────────────
   280→
   281→  useEffect(() => {
   282→    const lastId = localStorage.getItem('iptv_last_channel');
   283→    if (lastId) {
   284→      const ch = channels.find(c => c.id === parseInt(lastId, 10));
   285→      if (ch) playChannel(ch);
   286→    }
   287→  }, [channels, playChannel]);
   288→
   289→  // ─── Toggle favorite ────────────────────────────────────────────────────────
   290→
   291→  const toggleFavorite = useCallback((channelId: number, e: React.MouseEvent) => {
   292→    e.stopPropagation();
   293→    setFavorites(prev => {
   294→      const updated = prev.includes(channelId) ? prev.filter(id => id !== channelId) : [...prev, channelId];
   295→      localStorage.setItem('iptv_favorites', JSON.stringify(updated));
   296→      return updated;
   297→    });
   298→  }, []);
   299→
   300→  // ─── Prev/Next ──────────────────────────────────────────────────────────────
   301→
   302→  const prevChannel = useCallback(() => {
   303→    if (!currentChannel) return;
   304→    const idx = filteredChannels.findIndex(c => c.id === currentChannel.id);
   305→    if (idx > 0) playChannel(filteredChannels[idx - 1]);
   306→    else if (filteredChannels.length > 0) playChannel(filteredChannels[filteredChannels.length - 1]);
   307→  }, [currentChannel, filteredChannels, playChannel]);
   308→
   309→  const nextChannel = useCallback(() => {
   310→    if (!currentChannel) return;
   311→    const idx = filteredChannels.findIndex(c => c.id === currentChannel.id);
   312→    if (idx < filteredChannels.length - 1) playChannel(filteredChannels[idx + 1]);
   313→    else if (filteredChannels.length > 0) playChannel(filteredChannels[0]);
   314→  }, [currentChannel, filteredChannels, playChannel]);
   315→
   316→  const retryChannel = useCallback(() => {
   317→    if (currentChannel) playChannel(currentChannel);
   318→  }, [currentChannel, playChannel]);
   319→
   320→  // ─── Fullscreen ─────────────────────────────────────────────────────────────
   321→
   322→  const toggleFullscreen = useCallback(() => {
   323→    if (!containerRef.current) return;
   324→    if (document.fullscreenElement) {
   325→      document.exitFullscreen();
   326→    } else {
   327→      containerRef.current.requestFullscreen();
   328→    }
   329→  }, []);
   330→
   331→  useEffect(() => {
   332→    const handler = () => setIsFullscreen(!!document.fullscreenElement);
   333→    document.addEventListener('fullscreenchange', handler);
   334→    return () => document.removeEventListener('fullscreenchange', handler);
   335→  }, []);
   336→
   337→  // ─── PiP ────────────────────────────────────────────────────────────────────
   338→
   339→  const togglePiP = useCallback(async () => {
   340→    const video = videoRef.current;
   341→    if (!video) return;
   342→    try {
   343→      if (document.pictureInPictureElement) {
   344→        await document.exitPictureInPicture();
   345→        setIsPiP(false);
   346→      } else {
   347→        await video.requestPictureInPicture();
   348→        setIsPiP(true);
   349→      }
   350→    } catch { /* PiP not supported */ }
   351→  }, []);
   352→
   353→  // ─── Keyboard shortcuts ────────────────────────────────────────────────────
   354→
   355→  useEffect(() => {
   356→    const handleKeyDown = (e: KeyboardEvent) => {
   357→      if (showHelp || showImport) {
   358→        if (e.key === 'Escape') { setShowHelp(false); setShowImport(false); }
   359→        return;
   360→      }
   361→      if (e.target instanceof HTMLInputElement) return;
   362→      switch (e.key) {
   363→        case ' ':
   364→          e.preventDefault();
   365→          if (videoRef.current) videoRef.current.paused ? videoRef.current.play() : videoRef.current.pause();
   366→          break;
   367→        case 'm': case 'M': setMuted(p => !p); break;
   368→        case 'f': case 'F': toggleFullscreen(); break;
   369→        case 'ArrowUp': e.preventDefault(); setVolume(v => Math.min(1, v + 0.1)); break;
   370→        case 'ArrowDown': e.preventDefault(); setVolume(v => Math.max(0, v - 0.1)); break;
   371→        case 'ArrowLeft': if (!sidebarOpen && videoRef.current) videoRef.current.currentTime -= 10; break;
